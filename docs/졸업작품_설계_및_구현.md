def detect_phishing(email) -> PhishingResult:
    risk_score = 0
    
    # 1. 긴급한 언어 (최대 20점)
    if has_urgent_language(email.body):
        risk_score += 20
    
    # 2. 단축 URL (최대 25점)
    if has_shortened_url(email.body):
        risk_score += 25
    
    # 3. SPF/DKIM/DMARC 실패 (25점)
    if not check_email_auth(email):
        risk_score += 25
    
    # 4. From/Reply-To 불일치 (15점)
    if email.from_addr != email.reply_to:
        risk_score += 15
    
    # 5. 계정 정보 요청 (최대 25점)
    if requests_account_info(email.body):
        risk_score += 25
    
    # 위험 점수 ≥ 30: 피싱
    is_phishing = risk_score >= 30
    
    return PhishingResult(
        is_phishing=is_phishing,
        risk_score=risk_score,
        indicators=indicators
    )

    # 4. From/Reply-To nomuvofiqlik (15 ball)
    if email.from_addr != email.reply_to:
        risk_score += 15
    
    # 5. Hisob ma'lumotlari so'rash (maksimal 25 ball)
    if requests_account_info(email.body):
        risk_score += 25
    
    # Xavf balli ≥ 30: phishing
    is_phishing = risk_score >= 30
    
    return PhishingResult(
        is_phishing=is_phishing,
        risk_score=risk_score,
        indicators=indicators
    )
